if-else statements: 

		var number1 = 10;
		var number2 = 100;
		
		if (number1 > number2) {
			alert('true!');  <--the alert will not happen, 10 is NOT bigger than 100
		    } else {
			alert('false'); <--the else statement causes an alert for anyhting that is not true
		}
